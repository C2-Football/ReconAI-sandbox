# Target file: shared/gm-engine.js

## What You're Fixing and Why

The home page Field Intel bullets and Priority list are generated by two
functions in shared/gm-engine.js: `generateFieldIntel()` and
`generatePriorities()`. Both functions have real data access — they call
`_assess()`, `_myRoster()`, `_strategy()`, and `_LI()`. The data
plumbing is mostly correct.

The bugs are logical, not structural:

**Bug 1 — generatePriorities() ignores actual QB count**
Priority 1 in offseason mode reads `_topPremium[0]` — the first position
in the user's "needs" array. The needs array is driven by `assessTeam`,
which compares startable players (by PPG threshold) to roster slot count.
The problem: in the early offseason, PPG data is stale or zero (no games
played yet), so nearly every position flags as a "need" regardless of how
many players you have. A team with 6 QBs still gets "QB room needs a
foundational piece" as Priority 1.

**Bug 2 — generateFieldIntel() falls through to a hardcoded platitude**
Near the bottom of generateFieldIntel(), inside the
`mode === 'rebuild' || mode === 'balanced_rebuild'` block, there's a
fallback:
```js
if (obs.length < 3) {
  obs.push("Young WR/RB acquisitions compound over time — favor upside over short-term production.");
}
```
And in the health trend filler block at the very bottom:
```js
obs.push("Draft capital usage rate is below the league's top contenders.");
```
These are hardcoded strings. They are not derived from any data. They
appear on every team's home page regardless of roster composition.

**Bug 3 — Priority 1 offseason uses generic consequence text**
Even when the position detection is correct, the consequence text is
generic: "QB is the engine of dynasty rosters. Lock in your piece now."
This ignores DHQ scores of current players at that position, the gap
size, and league context.

## The Fixes

### Fix 1: Add a position count guard to generatePriorities()

Before pushing a positional priority in the offseason block, check that
the user doesn't already have adequate raw count at that position. Use
this logic:

```js
// Count players at position regardless of PPG (offseason has no PPG data)
function _rawCount(playerIds, pos) {
  return (playerIds || []).filter(pid => _pos(pid) === pos).length;
}
```

Then in the offseason Priority 1 block, replace the current push with:

```js
if (_topPremium.length > 0) {
  const pos = _topPremium[0];
  const rawCount = _rawCount(myRoster.players, pos);
  // In offseason, if they have 3+ players at the position, skip it
  // unless their best player's DHQ is below a starter threshold
  const posPlayers = _rosterPlayersAtPos(myRoster.players, pos);
  const bestDhq = posPlayers[0]?.dhq || 0;
  const starterThreshold = pos === 'QB' ? 3500 : pos === 'TE' ? 2500 : 3000;

  if (rawCount >= 3 && bestDhq >= starterThreshold) {
    // They're fine at this position — skip to next need
    const nextPos = _topPremium.find((p, i) => {
      if (i === 0) return false;
      const cnt = _rawCount(myRoster.players, p);
      const top = _rosterPlayersAtPos(myRoster.players, p)[0]?.dhq || 0;
      const thresh = p === 'QB' ? 3500 : p === 'TE' ? 2500 : 3000;
      return !(cnt >= 3 && top >= thresh);
    });
    if (nextPos) {
      // Push the actual gap position instead
      const posPlayersList = _rosterPlayersAtPos(myRoster.players, nextPos);
      const topDhq = posPlayersList[0]?.dhq || 0;
      const dhqNote = topDhq > 0 ? ` Your best ${nextPos} is DHQ ${topDhq.toLocaleString()}.` : '';
      priorities.push({
        problem: `${nextPos} room needs a foundational piece before the season`,
        consequence: `${nextPos} is your biggest positional gap.${dhqNote} Address now while offseason prices are lower.`,
        actionLabel: `Target ${nextPos}`,
        actionType: 'trade',
      });
    }
    // If no next pos either, skip Priority 1 positional entirely
  } else if (rawCount < 3 || bestDhq < starterThreshold) {
    // Genuine gap — push with DHQ context
    const posPlayersList = _rosterPlayersAtPos(myRoster.players, pos);
    const topDhq = posPlayersList[0]?.dhq || 0;
    const dhqNote = topDhq > 0
      ? ` Your best ${pos} is ${posPlayersList[0]?.name} at DHQ ${topDhq.toLocaleString()} — below starter threshold.`
      : ` You have no established ${pos} starter.`;
    const consequence = _isPPR && pos === 'WR'
      ? `PPR leagues reward deep WR rooms. Build now while prices are lower.${dhqNote}`
      : `${pos} is the engine of dynasty rosters.${dhqNote}`;
    priorities.push({
      problem: `${pos} room needs a foundational piece before the season`,
      consequence,
      actionLabel: `Target ${pos}`,
      actionType: 'trade',
    });
  }
}
```

### Fix 2: Replace the three hardcoded platitude strings in generateFieldIntel()

Find and remove these three hardcoded strings:

1. `"Young WR/RB acquisitions compound over time — favor upside over short-term production."`
2. `"Draft capital usage rate is below the league's top contenders."`
3. `"Win-now window requires decisive action — indecision is its own risk."` (generic fallback)

Replace the fallback blocks with data-derived observations. Here is the
replacement logic for the rebuild fallback (currently `if (obs.length < 3)`
inside the rebuild block):

```js
if (obs.length < 3) {
  // Generate a DHQ-grounded observation instead of a platitude
  const myPlayers = myRoster?.players || [];
  const wrPlayers = _rosterPlayersAtPos(myPlayers, 'WR');
  const rbPlayers = _rosterPlayersAtPos(myPlayers, 'RB');

  // Find youngest high-upside players (age < 24, DHQ > 2000)
  const S = _S();
  const youngAssets = myPlayers
    .map(pid => ({
      pid,
      name: _name(pid),
      pos: _pos(pid),
      dhq: _dhq(pid),
      age: S.players?.[pid]?.age || 99,
    }))
    .filter(p => p.age < 24 && p.dhq > 2000 && ['WR','RB','QB','TE'].includes(p.pos))
    .sort((a, b) => b.dhq - a.dhq);

  if (youngAssets.length >= 2) {
    obs.push(`${youngAssets[0].name} and ${youngAssets[1].name} are your young cornerstones — both under 24 with long peak windows ahead.`);
  } else if (youngAssets.length === 1) {
    obs.push(`${youngAssets[0].name} (age ${youngAssets[0].age}, DHQ ${youngAssets[0].dhq.toLocaleString()}) is your best young asset — protect this piece.`);
  } else {
    // No young high-value players — that's an actual insight
    obs.push('Your roster skews veteran — prioritize youth in the rookie draft to extend your contention window.');
  }
}
```

Replace the health trend filler block at the bottom (the one that pushes
`"Draft capital usage rate is below the league's top contenders."`) with:

```js
if (obs.length < 3) {
  // Compare pick inventory to league average — data driven
  const S = _S();
  const allTP = S.tradedPicks || [];
  const myId = myRoster?.roster_id;
  const curYear = parseInt(S.season) || new Date().getFullYear();

  if (myId) {
    // My future picks
    const myPicks = allTP.filter(p =>
      p.owner_id === myId &&
      parseInt(p.season) >= curYear
    ).length;

    // League average future picks
    const rosters = S.rosters || [];
    if (rosters.length > 1) {
      const otherPickCounts = rosters
        .filter(r => r.roster_id !== myId)
        .map(r => allTP.filter(p => p.owner_id === r.roster_id && parseInt(p.season) >= curYear).length);
      const leagueAvg = otherPickCounts.length
        ? Math.round(otherPickCounts.reduce((a, b) => a + b, 0) / otherPickCounts.length)
        : 0;

      if (leagueAvg > 0) {
        const diff = myPicks - leagueAvg;
        if (diff >= 2) {
          obs.push(`You hold ${myPicks} future picks — ${diff} more than the league average. Strong draft capital position.`);
        } else if (diff <= -2) {
          obs.push(`You hold ${myPicks} future picks vs. a league average of ${leagueAvg}. Draft capital is below average — consider pick acquisition.`);
        } else {
          obs.push(`Your pick inventory (${myPicks}) is in line with the league average (${leagueAvg}).`);
        }
      } else if (myPicks > 0) {
        obs.push(`You currently hold ${myPicks} future draft picks.`);
      }
    }
  }
}
```
